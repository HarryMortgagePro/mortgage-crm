Prisma Schema – Core Models

Create these Prisma models (you can adjust field names slightly if idiomatic, but keep all fields and relations):

model Client {
  id             Int              @id @default(autoincrement())
  firstName      String
  lastName       String
  email          String?
  phone          String?
  tags           String?          // comma-separated tags: "first-time, investor"
  referralSource String?          // Realtor, Google, Instagram, Friend, etc.
  notes          String?
  applications   Application[]
  tasks          Task[]
  createdAt      DateTime         @default(now())
  updatedAt      DateTime         @updatedAt
}

model Application {
  id                 Int              @id @default(autoincrement())
  client             Client           @relation(fields: [clientId], references: [id])
  clientId           Int

  applicationType    String           // Mortgage, HELOC, Personal, Auto, Business, Private, Construction
  purpose            String           // Purchase, Refinance, Renewal, Pre-Approval, Equity Take-Out, Bridge
  status             String           // Lead, Pre-Qualified, Submitted, Conditional Approval, Docs Pending, Compliance Review, Final Approval, Signed, Funded, Declined, Withdrawn

  propertyAddress    String?
  propertyCity       String?
  propertyProvince   String?
  propertyType       String?          // Condo, Detached, Townhouse, Multi-unit, Commercial

  purchasePrice      Float?
  mortgageAmount     Float?
  downPaymentAmount  Float?
  interestRate       Float?
  rateType           String?          // Fixed, Variable
  termYears          Int?
  amortizationYears  Int?

  lenderName         String?
  brokerFee          Float?           // just store a number (can be % or flat)
  lenderFee          Float?

  applicationDate    DateTime?
  conditionsDueDate  DateTime?
  closingDate        DateTime?
  fundedDate         DateTime?

  notes              String?

  tasks              Task[]
  documents          DocumentRecord[]

  createdAt          DateTime         @default(now())
  updatedAt          DateTime         @updatedAt
}

model Task {
  id             Int         @id @default(autoincrement())
  title          String
  description    String?
  dueDate        DateTime?
  status         String      // Open, In Progress, Done
  priority       String?     // Low, Medium, High
  category       String?     // Documents, Follow-up, Compliance, Lender, Other

  client         Client?     @relation(fields: [clientId], references: [id])
  clientId       Int?

  application    Application? @relation(fields: [applicationId], references: [id])
  applicationId  Int?

  createdAt      DateTime    @default(now())
  updatedAt      DateTime    @updatedAt
}

model DocumentRecord {
  id             Int         @id @default(autoincrement())
  application    Application @relation(fields: [applicationId], references: [id])
  applicationId  Int
  docType        String      // T1, NOA, T4, Job Letter, Bank Statements, APS, Appraisal, etc.
  status         String      // Requested, Received, Expired, Not Required
  notes          String?

  createdAt      DateTime    @default(now())
  updatedAt      DateTime    @updatedAt
}
